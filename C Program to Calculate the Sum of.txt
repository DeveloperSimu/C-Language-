// C Program to Calculate the Sum of Squares of First N Natural Numbers
#include<stdio.h>
int main()
{int n, i, sum=0;
printf("\n Enter the limit value:");
scanf("%d",&n);
for(i=1;i<=n;i++)
{sum=sum+(i*i);
}
printf("\n The summation of the given series is=%d", sum);
return 0;
}
