// C Program to Find Maximum, Minimum and Their Sum in an Array of 10 Elements

#include<stdio.h>
int main()
{
    int n[10],i,max,min,sum;
    printf("\n enter 10 values:");
    for(i=0;i<10;i++)
    {
        scanf("%d",&n[i]);
    }
    min=max=n[0];
    for(i=1;i<10;i++)
{
    {
        if(n[i]>max)
        max=n[i];
    }
    {
        if(n[i]<min)
        min=n[i];
    }
}
sum=(max+min);
printf("\n maximun number is=%d",max);
printf("\n minimum number is=%d", min);
printf("\n sum=%d",sum);

}
