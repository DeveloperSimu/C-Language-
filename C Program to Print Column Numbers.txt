// C Program to Print Column Numbers in a Square Pattern
#include<stdio.h>
int main()
{
int n,i,j ;
printf("Enter number of rows : ");
scanf("%d",&n);
for(i=1;i<=n;i++)
{for(j=1;j<=n;j++)
{printf(" %d",j);
}
printf("\n");
}
return 0;
}
