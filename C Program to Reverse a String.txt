// C Program to Reverse a String

#include <stdio.h>
#include <string.h>

int main() {
    char str[20], revstr[20];
    int i, j = 0, n;

    printf("Enter a string: ");
    scanf("%s", str);  // safer than gets()

    n = strlen(str);

    for(i = n - 1; i >= 0; i--) {
        revstr[j] = str[i];
        j++;
    }
    revstr[j] = '\0';

    printf("The reversed string is: %s\n", revstr);

    return 0;
}


Output:
Enter a string: hello
The reversed string is: olleh
