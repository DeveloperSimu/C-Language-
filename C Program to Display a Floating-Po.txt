// C Program to Display a Floating-Point Value using Different printf() Formats

#include<stdio.h>
int main()
{float a=98.7628;
printf("\n The float number is=%f", a);
printf("\n The float number is=%7.4f", a);
printf("\n The float number is=%7.2f", a);
printf("\n The float number is=%10.2e", a);
printf("\n The float number is=%11.4e", a);
printf("\n The float number is=%-10.2e", a);
printf("\n The float number is=%e", a);
return 0;
}
