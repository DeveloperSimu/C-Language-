// C Program to Calculate the Sum of a Nested Series

#include<stdio.h>
int main()
{
    int n, sum, sum1 = 0, i, j;
    printf("\n Enter limit value: ");
    scanf("%d", &n);

    for(i = 1; i <= n; i++)
    {
        sum = 0;
        for(j = 1; j <= i; j++)
        {
            sum += j;
        }
        sum1 += sum;
    }

    printf("\nThe sum of the series up to %d terms is: %d\n", n, sum1);
    return 0;
}


Output:
Enter limit value: 4
The sum of the series up to 4 terms is: 20
