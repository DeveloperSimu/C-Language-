// C Program to Calculate the Sum of the Series x + x^2/2! + x^3/3! + ... up to n terms

#include<stdio.h>
#include<math.h>
int main()
{
 float x=2,n, i,sum=0.0,fact=1;
 printf("\nEnter the limit value:");
 scanf("%f", &n);
 for(i=1;i<=n;i++)
 {
 fact=fact*i;
  sum=sum+pow(x,i)/fact;
 }
 printf("\n The Sum of the given series is=%f", sum);
 return 0;
}


Output:
Enter the limit value: 5
The Sum of the given series is=6.333333
